// =====================================================
// TABLAS PRINCIPALES (ORDEN LÓGICO)
// =====================================================

// 1. TABLA DE USUARIOS (Base del sistema)
Table Usuarios {
  Id int [pk, increment]
  EntidadId int [not null]
  Nombre varchar(100) [not null]
  Apellidos varchar(100) [not null]
  Correo varchar(255) [not null, unique]
  Password varchar(255) [not null]
  Telefono varchar(20)
  Identificacion varchar(20) [unique]
  Avatar varchar(500)
  IpPublica varchar(45) [note: 'IPv4 o IPv6 para validación de ban']
  Rol varchar(50) [not null, default: 'Usuario']
  IsActive bit [not null, default: 1]
  IsBanned bit [not null, default: 0]
  FechaBan datetime2 [note: 'Fecha cuando fue baneado']
  MotivoBan varchar(500) [note: 'Motivo del ban']
  CreatedAt datetime2 [default: `GETDATE()`]
  UpdatedAt datetime2 [default: `GETDATE()`]
}

// 2. TABLA DE JUEGOS (Catálogo de juegos disponibles)
Table Juegos {
  Id int [pk, increment]
  Nombre varchar(100) [not null]
  Descripcion varchar(500)
  Categoria varchar(50)
  Dificultad varchar(50)
  MaxJugadores int [not null]
  Titulares int [not null]
  Suplentes int [not null]
  Icon varchar(500)
  Logo varchar(500)
  IsActive bit [not null, default: 1]
  CreatedAt datetime2 [default: `GETDATE()`]
  UpdatedAt datetime2 [default: `GETDATE()`]
}

// 3. TABLA DE PALETAS DE COLORES POR JUEGO
Table PaletasJuego {
  Id int [pk, increment]
  JuegoId int [not null, ref: > Juegos.Id]
  PrimaryColor varchar(7) [not null]
  SecondaryColor varchar(7) [not null]
  TertiaryColor varchar(7) [not null]
  AccentColor varchar(7) [not null]
  LightColor varchar(7) [not null]
}

// 4. TABLA DE JUGADORES (Perfiles de juego de usuarios)
Table Jugadores {
  Id int [pk, increment]
  UsuarioId int [not null, ref: > Usuarios.Id]
  Nombre varchar(100) [not null]
  Role varchar(50)
  Experiencia int [default: 0]
  Avatar varchar(500)
  IsAvailable bit [not null, default: 1]
  CreatedAt datetime2 [default: `GETDATE()`]
}

// 5. TABLA DE EQUIPOS (Creados por jugadores)
Table Equipos {
  Id int [pk, increment]
  Nombre varchar(100) [not null]
  Descripcion varchar(500)
  Logo varchar(500)
  Imagen varchar(500)
  Tag varchar(10)
  CreadorId int [not null, ref: > Jugadores.Id]
  IsActive bit [not null, default: 1]
  CreatedAt datetime2 [default: `GETDATE()`]
  UpdatedAt datetime2 [default: `GETDATE()`]
}

// 6. TABLA DE RELACIÓN JUGADOR-EQUIPO (Muchos a muchos)
Table JugadorEquipo {
  Id int [pk, increment]
  JugadorId int [not null, ref: > Jugadores.Id]
  EquipoId int [not null, ref: > Equipos.Id]
  EsCapitan bit [not null, default: 0]
  EsTitular bit [not null, default: 0]
  FechaUnion datetime2 [default: `GETDATE()`]
  IsActive bit [not null, default: 1]
  
  indexes {
    (JugadorId, EquipoId) [unique]
  }
}

// 7. TABLA DE ESTADÍSTICAS DE JUGADORES
Table EstadisticasJugador {
  Id int [pk, increment]
  JugadorId int [not null, ref: > Jugadores.Id]
  Kills int [default: 0]
  Deaths int [default: 0]
  Assists int [default: 0]
  WinRate decimal(5,2) [default: 0]
  GamesPlayed int [default: 0]
  BestGame varchar(100)
  Streak int [default: 0]
  Tier varchar(20) [default: 'bronze']
  Points int [default: 0]
  Level int [default: 1]
  Experience int [default: 0]
  ExperienceToNext int [default: 100]
  TotalPoints int [default: 0]
}

// 8. TABLA DE ESTADÍSTICAS DE EQUIPOS
Table EstadisticasEquipo {
  Id int [pk, increment]
  EquipoId int [not null, ref: > Equipos.Id]
  TotalMatches int [default: 0]
  Wins int [default: 0]
  Losses int [default: 0]
  Rank varchar(50)
  Tier varchar(20) [default: 'bronze']
  Points int [default: 0]
}

// 9. TABLA DE REQUISITOS DE JUEGO POR EQUIPO
Table RequisitosJuegoEquipo {
  Id int [pk, increment]
  EquipoId int [not null, ref: > Equipos.Id]
  JuegoId int [not null, ref: > Juegos.Id]
  Titulares int [not null]
  Suplentes int [not null]
  TotalJugadores int [not null]
}

// =====================================================
// TABLAS DE TORNEOS
// =====================================================

// 10. TABLA DE TORNEOS
Table Torneos {
  Id int [pk, increment]
  Nombre varchar(100) [not null]
  JuegoId int [not null, ref: > Juegos.Id]
  Descripcion varchar(500)
  FechaInicio datetime2 [not null]
  FechaFin datetime2 [not null]
  MaxParticipantes int [not null]
  Premio varchar(100)
  Estado varchar(20) [default: 'upcoming']
  Categoria varchar(50)
  Dificultad varchar(50)
  Ubicacion varchar(100)
  Formato varchar(50) [default: 'single_elimination']
  CostoEntrada decimal(10,2) [default: 0]
  Imagen varchar(500)
  Reglas text
  IsActive bit [not null, default: 1]
  CreatedAt datetime2 [default: `GETDATE()`]
  UpdatedAt datetime2 [default: `GETDATE()`]
}

// 11. TABLA DE PARTICIPANTES EN TORNEOS
Table ParticipantesTorneo {
  Id int [pk, increment]
  TorneoId int [not null, ref: > Torneos.Id]
  EquipoId int [ref: > Equipos.Id]
  JugadorId int [ref: > Jugadores.Id]
  Seed int
  Logo varchar(500)
  Icon varchar(10)
  FechaRegistro datetime2 [default: `GETDATE()`]
  Estado varchar(20) [default: 'registered']
  
  indexes {
    (EquipoId, JugadorId) [note: 'Un participante debe ser equipo O jugador, no ambos']
  }
}

// 12. TABLA DE PARTIDAS
Table Partidas {
  Id int [pk, increment]
  TorneoId int [not null, ref: > Torneos.Id]
  Ronda int [not null]
  NumeroPartida int [not null]
  Participante1Id int [not null, ref: > ParticipantesTorneo.Id]
  Participante2Id int [not null, ref: > ParticipantesTorneo.Id]
  PuntuacionParticipante1 int [default: 0]
  PuntuacionParticipante2 int [default: 0]
  GanadorId int [ref: > ParticipantesTorneo.Id]
  Estado varchar(20) [default: 'pending']
  FechaProgramada datetime2
  Duracion varchar(20)
  Venue varchar(100)
  Notas varchar(500)
  CreatedAt datetime2 [default: `GETDATE()`]
}

// 13. TABLA DE RESULTADOS DETALLADOS DE PARTIDAS
Table ResultadosPartida {
  Id int [pk, increment]
  PartidaId int [not null, ref: > Partidas.Id]
  PuntosParticipante1 int [default: 0]
  PuntosParticipante2 int [default: 0]
  DiferenciaMapasParticipante1 int [default: 0]
  DiferenciaMapasParticipante2 int [default: 0]
  DiferenciaRondasParticipante1 int [default: 0]
  DiferenciaRondasParticipante2 int [default: 0]
  VictoriaPerfecta bit [default: 0]
}

// 14. TABLA DE ESTADÍSTICAS DE JUGADORES POR PARTIDA
Table EstadisticasJugadorPartida {
  Id int [pk, increment]
  PartidaId int [not null, ref: > Partidas.Id]
  JugadorId int [not null, ref: > Jugadores.Id]
  Puntos int [default: 0]
  Kills int [default: 0]
  Assists int [default: 0]
  Deaths int [default: 0]
  KDA decimal(5,2) [default: 0]
  RondasGanadas int [default: 0]
  RondasJugadas int [default: 0]
}

}

// 19. TABLA DE CONFIGURACIÓN DE NOTIFICACIONES POR USUARIO
Table ConfiguracionNotificaciones {
  Id int [pk, increment]
  UsuarioId int [not null, ref: > Usuarios.Id]
  NotificacionesPlataforma bit [not null, default: 1]
  NotificacionesEmail bit [not null, default: 1]
  NotificacionesWhatsApp bit [not null, default: 0]
  NotificacionesPush bit [not null, default: 1]
  NotificacionesTorneos bit [not null, default: 1]
  NotificacionesEquipos bit [not null, default: 1]
  NotificacionesLogros bit [not null, default: 1]
  NotificacionesSistema bit [not null, default: 1]
  HoraInicioNotificaciones time [default: '08:00']
  HoraFinNotificaciones time [default: '22:00']
  ZonaHoraria varchar(50) [default: 'America/Mexico_City']
  IsActive bit [not null, default: 1]
  CreatedAt datetime2 [default: `GETDATE()`]
  UpdatedAt datetime2 [default: `GETDATE()`]
}

// 20. TABLA DE NOTIFICACIONES
Table Notificaciones {
  Id int [pk, increment]
  UsuarioId int [not null, ref: > Usuarios.Id]
  ConfiguracionNotificacionId int [ref: > ConfiguracionNotificaciones.Id]
  Titulo varchar(100) [not null]
  Mensaje varchar(500) [not null]
  Tipo varchar(20) [not null]
  Categoria varchar(50)
  CanalEnvio varchar(20) [default: 'plataforma']
  EstadoEnvio varchar(20) [default: 'pendiente']
  FechaEnvio datetime2
  IsRead bit [not null, default: 0]
  FechaCreacion datetime2 [default: `GETDATE()`]
  FechaLectura datetime2
}

// 21. TABLA DE CONFIGURACIÓN DEL SISTEMA
